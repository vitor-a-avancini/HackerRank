def flippingBits(n):
    # Write your code here
    b = '{:032b}'.format(n)
    stra = ''+b
    rep=stra.replace('0','W')
    rep=rep.replace('1','0')
    rep=rep.replace('W','1')
    b = int(rep, 2)
    return b
